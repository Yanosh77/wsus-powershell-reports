function Resolve-Lang([string]$l){
  if([string]::IsNullOrWhiteSpace($l) -or $l -eq 'auto'){
    try { $l = [System.Globalization.CultureInfo]::CurrentUICulture.TwoLetterISOLanguageName } catch { $l = 'en' }
  }
  switch ($l.ToLower()){
    'fr' { 'fr' }
    'es' { 'es' }
    default { 'en' }
  }
}
